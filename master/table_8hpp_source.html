166 template<class Tab>
167 void append_rows(Tab&& other,
168 const Size count = std::numeric_limits<Size>::max())
169 {
170 static_assert(std::is_same_v<std::decay_t<Tab>, Table>);
171
172 if (!column_count()) {
173 columns_ = {}; // prevent UB if instance was moved
174 columns_.resize(other.column_count());
175 } else if (!(column_count() == other.column_count()))
176 throw Exception{"cannot append table rows from table with different "
177 "column count"};
178
179 const Size cc = column_count();
180 const Size in_size = std::min(other.row_count(), count);
181 const Size out_offset = row_count();
182 for (Size i{}; i < cc; ++i) {
183 columns_[i].resize(out_offset + in_size);
184 const auto b = other.columns_[i].begin();
185 const auto e = b + in_size;
186 const auto o = columns_[i].begin() + out_offset;
187 if constexpr (std::is_rvalue_reference_v<decltype(std::forward<Tab>(other))>)
188 std::move(b, e, o);
189 else
190 std::copy(b, e, o);
191 }
192 }

231 template<class C>
232 void append_column(C&& column)
233 {
234 static_assert(std::is_same_v<std::decay_t<C>, Column>);
235
236 if (!(!column_count() || (row_count() == column.size())))
237 throw Exception{"cannot append table column with different row count"};
238
239 columns_.push_back(std::forward<C>(column));
