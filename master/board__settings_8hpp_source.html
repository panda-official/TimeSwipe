37enum class Measurement_mode {

46constexpr std::optional<Measurement_mode> to_measurement_mode(const
47 std::string_view value) noexcept
48{
49 if (value == "voltage") return Measurement_mode::voltage;
50 else if (value == "current") return Measurement_mode::current;
51 else return {};
52}
53
58constexpr const char* to_literal(const Measurement_mode value) noexcept
59{
60 switch (value) {
61 case Measurement_mode::voltage: return "voltage";
62 case Measurement_mode::current: return "current";
63 }
64 return nullptr;
65}
