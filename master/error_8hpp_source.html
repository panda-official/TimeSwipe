128constexpr const char* to_literal_anyway(const Errc errc) noexcept
129{
130 constexpr const char* unknown{"unknown error"};

164class Generic_error_category final : public std::error_category {
165public:
167 const char* name() const noexcept override
168 {
169 return "panda_timeswipe_generic_error";
170 }
171
181 std::string message(const int ev) const override
182 {
183 const char* const desc{to_literal_anyway(static_cast<Errc>(ev))};
184 constexpr const char* const sep{": "};
185 std::string result;
186 result.reserve(std::strlen(name()) + std::strlen(sep) + std::strlen(desc));
187 return result.append(name()).append(sep).append(desc);
188 }
189};
190
196inline const Generic_error_category& generic_error_category() noexcept
197{
198 static const Generic_error_category instance;
199 return instance;
200}
201
207inline std::error_condition make_error_condition(const Errc errc) noexcept
208{
209 return {static_cast<int>(errc), generic_error_category()};
210}
